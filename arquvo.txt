def busca_binaria_matriz(matriz, item):
    linhas = len(matriz)
    colunas = len(matriz[0])
    
    baixo = 0
    alto = linhas * colunas - 1
    
    while baixo <= alto:
        meio = (baixo + alto) // 2
        meio_valor = matriz[meio // colunas][meio % colunas]
        
        if meio_valor == item:
            return (meio // colunas, meio % colunas)
        elif meio_valor < item:
            baixo = meio + 1
        else:
            alto = meio - 1
    
    return None
